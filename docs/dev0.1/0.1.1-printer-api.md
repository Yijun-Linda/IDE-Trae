## 1. 技术调研

### 1.1 易联云API

**基础信息**
- 官网: https://www.10ss.net/
- API文档: https://www.10ss.net/help.html
- 接口地址: https://open-api.10ss.net/
- 认证方式: sign签名（md5）

**核心接口**

| 接口 | 路径 | 说明 |
|------|------|------|
| 添加打印机 | /api/addprinter | 绑定打印机到账号 |
| 打印文本 | /api/print | 发送打印内容 |
| 查询状态 | /api/queryPrinterStatus | 查询打印机在线状态 |
| 查询订单 | /api/queryOrderState | 查询打印任务状态 |

**签名算法**
```python
def generate_sign(params: dict, app_secret: str) -> str:
    """
    易联云签名算法
    1. 将所有参数按key排序
    2. 拼接成 key1value1key2value2... 格式
    3. 末尾加上 app_secret
    4. MD5加密，转大写
    """
    sorted_params = sorted(params.items())
    sign_str = ''.join([f"{k}{v}" for k, v in sorted_params]) + app_secret
    return hashlib.md5(sign_str.encode()).hexdigest().upper()
```

## 3. 接口规范

### 3.1 BasePrinter抽象基类

```python
from abc import ABC, abstractmethod
from typing import Optional

class BasePrinter(ABC):
    """打印机抽象基类"""
    
    def __init__(self, config: dict):
        self.config = config
        self.timeout = config.get("timeout", 10)
    
    @abstractmethod
    def print_ticket(self, content: str, dry_run: bool = False) -> dict:
        """
        打印小票
        
        Args:
            content: 打印内容（纯文本或ESC/POS指令）
            dry_run: 是否仅预览不实际打印
            
        Returns:
            {
                "success": bool,
                "order_id": str,      # 打印任务ID
                "message": str,       # 状态信息
                "preview": str        # dry_run时的预览内容
            }
        """
        pass
    
    @abstractmethod
    def query_status(self, order_id: str) -> dict:
        """
        查询打印任务状态
        
        Returns:
            {
                "success": bool,
                "status": str,        # pending/printing/completed/failed
                "printed": bool,      # 是否已打印
                "message": str
            }
        """
        pass
    
    @abstractmethod
    def is_online(self) -> bool:
        """检查打印机是否在线"""
        pass
    
    def validate_content(self, content: str) -> bool:
        """验证打印内容（通用检查）"""
        if not content:
            return False
        if len(content) > 4096:  # 最大长度限制
            return False
        return True
```

## 4. 实现细节

### 4.1 易联云实现

```python
import hashlib
import time
import requests
from typing import Optional

class YilianyunPrinter(BasePrinter):
    """易联云打印机实现"""
    
    API_BASE = "https://open-api.10ss.net"
    
    def __init__(self, config: dict):
        super().__init__(config)
        self.app_id = config["app_id"]
        self.app_secret = config["app_secret"]
        self.machine_code = config["machine_code"]
    
    def _generate_sign(self, params: dict) -> str:
        """生成签名"""
        sorted_params = sorted(params.items())
        sign_str = ''.join([f"{k}{v}" for k, v in sorted_params]) + self.app_secret
        return hashlib.md5(sign_str.encode()).hexdigest().upper()
    
    def _call_api(self, endpoint: str, params: dict) -> dict:
        """调用API"""
        url = f"{self.API_BASE}{endpoint}"
        params["appid"] = self.app_id
        params["timestamp"] = str(int(time.time()))
        params["sign"] = self._generate_sign(params)
        
        try:
            response = requests.post(url, data=params, timeout=self.timeout)
            return response.json()
        except Exception as e:
            return {"error": str(e)}
    
    def print_ticket(self, content: str, dry_run: bool = False) -> dict:
        """打印小票"""
        if not self.validate_content(content):
            return {"success": False, "message": "Invalid content"}
        
        if dry_run:
            return {
                "success": True,
                "order_id": None,
                "message": "Dry run mode",
                "preview": content
            }
        
        params = {
            "machine_code": self.machine_code,
            "content": content,
            "origin_id": f"order_{int(time.time() * 1000)}"
        }
        
        result = self._call_api("/api/print", params)
        
        return {
            "success": result.get("error") == "0",
            "order_id": result.get("id"),
            "message": result.get("error_description", "Unknown"),
            "preview": None
        }
    
    def query_status(self, order_id: str) -> dict:
        """查询打印状态"""
        params = {
            "machine_code": self.machine_code,
            "order_id": order_id
        }
        
        result = self._call_api("/api/queryOrderState", params)
        
        return {
            "success": result.get("error") == "0",
            "status": result.get("data", {}).get("status", "unknown"),
            "printed": result.get("data", {}).get("status") == "completed",
            "message": result.get("error_description", "")
        }
    
    def is_online(self) -> bool:
        """检查打印机是否在线"""
        params = {"machine_code": self.machine_code}
        result = self._call_api("/api/queryPrinterStatus", params)
        return result.get("error") == "0" and result.get("data", {}).get("status") == "online"
```

### 4.2 飞鹅云实现

```python
import hashlib
import time
import requests

class FeiePrinter(BasePrinter):
    """飞鹅云打印机实现"""
    
    API_BASE = "https://api.feieyun.com"
    
    def __init__(self, config: dict):
        super().__init__(config)
        self.user = config["user"]
        self.ukey = config["ukey"]
        self.sn = config["sn"]
    
    def _generate_sign(self, stime: str) -> str:
        """生成签名"""
        sign_str = self.user + self.ukey + stime
        return hashlib.sha1(sign_str.encode()).hexdigest()
    
    def _call_api(self, apiname: str, params: dict) -> dict:
        """调用API"""
        url = f"{self.API_BASE}/Api/Open/"
        
        stime = str(int(time.time()))
        params.update({
            "user": self.user,
            "stime": stime,
            "sig": self._generate_sign(stime),
            "apiname": apiname
        })
        
        try:
            response = requests.post(url, data=params, timeout=self.timeout)
            return response.json()
        except Exception as e:
            return {"error": str(e)}
    
    def print_ticket(self, content: str, dry_run: bool = False) -> dict:
        """打印小票"""
        if not self.validate_content(content):
            return {"success": False, "message": "Invalid content"}
        
        if dry_run:
            return {
                "success": True,
                "order_id": None,
                "message": "Dry run mode",
                "preview": content
            }
        
        params = {
            "sn": self.sn,
            "content": content,
            "times": "1"
        }
        
        result = self._call_api("Open_printMsg", params)
        
        return {
            "success": result.get("ret") == 0,
            "order_id": result.get("data"),
            "message": result.get("msg", "Unknown"),
            "preview": None
        }
    
    def query_status(self, order_id: str) -> dict:
        """查询打印状态"""
        params = {"sn": self.sn, "orderid": order_id}
        result = self._call_api("Open_queryOrderState", params)
        
        return {
            "success": result.get("ret") == 0,
            "status": "completed" if result.get("data") == "1" else "pending",
            "printed": result.get("data") == "1",
            "message": result.get("msg", "")
        }
    
    def is_online(self) -> bool:
        """检查打印机是否在线"""
        params = {"sn": self.sn}
        result = self._call_api("Open_queryPrinterStatus", params)
        return result.get("ret") == 0 and "在线" in result.get("data", "")
```

## 6. 故障排查

### 6.1 常见问题

| 问题 | 可能原因 | 解决方案 |
|------|---------|---------|
| 签名错误 | 时间戳不同步 | 检查服务器时间，确保与NTP同步 |
| 打印机离线 | 网络/设备问题 | 检查打印机WiFi连接，重启设备 |
| 打印内容乱码 | 编码问题 | 使用GBK编码，避免特殊字符 |
| API限流 | 调用频率过高 | 实现指数退避重试机制 |
